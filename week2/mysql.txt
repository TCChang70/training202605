select ofc.officeCode,ofc.city,ofc.phone,ofc.addressLine1,
       emp.employeeNumber,emp.firstName,emp.extension,emp.email,emp.jobTitle
from classicmodels.offices as ofc
inner join classicmodels.employees as emp
on ofc.officeCode=emp.officeCode
order by ofc.officeCode


select emp.employeeNumber,emp.firstName,emp.extension,emp.email,
       cst.customerNumber,cst.customerName,cst.contactFirstName,cst.phone
from classicmodels.employees as emp
right join classicmodels.customers as cst
on emp.employeeNumber=cst.salesRepEmployeeNumber
where emp.employeeNumber is null


select cst.customerNumber,cst.contactFirstName,cst.phone,
       ord.orderNumber,ord.requiredDate,ord.status,
       det.productCode,det.priceEach
from classicmodels.customers as cst
inner join classicmodels.orders as ord
on cst.customerNumber=ord.customerNumber
inner join classicmodels.orderdetails as det
on ord.orderNumber=det.orderNumber
